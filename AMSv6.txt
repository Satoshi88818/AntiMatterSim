""" nuclear_evaporation.py ====================== Level 6: Nuclear evaporation model for antiproton annihilation on a bound nucleon within a nucleus ("nuclear star"). Physics ------- When an antiproton annihilates on a *bound* nucleon, the pion shower deposits energy into the residual nucleus, which then de-excites by evaporating light fragments: neutrons, protons, α-particles, deuterons, and occasionally a heavy recoil nucleus. Model summary (Weisskopf-Ewing statistical model): 1. The primary p̄+n̄ annihilation produces the same pion shower as the free-proton model (via AnnihilationModel). 2. The residual excitation energy (E*) deposited in the nucleus is sampled from a uniform distribution between 0 and the total pion KE. 3. Evaporated fragments are sampled with Weisskopf emission widths: Γ_f ∝ (2s_f + 1) · m_f · σ_inv(ε) · exp(-ε / T) where T = sqrt(E*/a), a = A/10 MeV⁻¹ (level-density parameter). 4. Emission continues until E* < E_threshold (≈ 8 MeV). 5. Remaining excitation is released as γ-rays (modelled as photon deposits). Nucleus database ---------------- Three target nuclei are supported: lead (Pb-208), carbon (C-12), oxygen (O-16). Medical relevance ----------------- Alpha particles and deuterons have very high LET (10–100 MeV·cm²/g) and sub-millimetre range at Bragg-peak energies. The biological dose enhancement from these fragments at the tumour site is the key motivation for antiproton cancer therapy (ACTA). References ---------- R.K. Bhaduri, L.E. Bhatt, J.C. Waddington (1973). Nuclear Evaporation. CERN Yellow Report on Antiproton Interactions (2004). """ from __future__ import annotations import numpy as np from dataclasses import dataclass from typing import List, Tuple from src.particles import ( PROTON_MASS_MEV, NEUTRON_MASS_MEV, ALPHA_MASS_MEV, DEUTERON_MASS_MEV, ALPHA, DEUTERON, HEAVY_RECOIL, PHOTON, ELECTRON_MASS_MEV, ) # --------------------------------------------------------------------------- # Nucleus database: (A, Z, binding_energy_mev, separation_energy_mev) # --------------------------------------------------------------------------- _NUCLEUS_DB = { "lead": {"A": 208, "Z": 82, "E_bind": 1636.4, "S_n": 7.37, "S_p": 8.00, "S_a": 9.72, "S_d": 11.5}, "carbon": {"A": 12, "Z": 6, "E_bind": 92.16, "S_n": 18.72, "S_p": 15.96, "S_a": 7.37, "S_d": 20.8}, "oxygen": {"A": 16, "Z": 8, "E_bind": 127.62, "S_n": 15.66, "S_p": 12.13, "S_a": 7.16, "S_d": 18.3}, } # Fragment emission channels: (ptype_code, mass_MeV, separation_key, spin_deg2s1) _CHANNELS = [ (ALPHA, ALPHA_MASS_MEV, "S_a", 1), # α (spin-0) (DEUTERON, DEUTERON_MASS_MEV, "S_d", 3), # d (spin-1) ] _EVAP_THRESHOLD_MEV = 8.0 # stop evaporation below this excitation _GAMMA_THRESHOLD_MEV = 2.0 # residual below this → γ deposit @dataclass class EvaporationEvent: """Fragments emitted by nuclear de-excitation.""" fragment_types: List[int] # particle type codes fragment_energies: List[float] # kinetic energy [MeV] fragment_directions: List[np.ndarray] # unit vectors gamma_energy_mev: float # residual γ deposit class NuclearEvaporationModel: """ Weisskopf-Ewing nuclear evaporation for p̄-nucleus annihilation. Parameters ---------- target_nucleus : str One of "lead", "carbon", "oxygen". rng : np.random.Generator, optional Shared RNG instance; created from seed if None. seed : int Seed used when rng is None. """ def __init__( self, target_nucleus: str = "lead", rng: np.random.Generator | None = None, seed: int = 0, ): if target_nucleus not in _NUCLEUS_DB: raise ValueError( f"Unknown nucleus '{target_nucleus}'. " f"Choices: {list(_NUCLEUS_DB)}" ) self._nuc = _NUCLEUS_DB[target_nucleus] self._rng = rng if rng is not None else np.random.default_rng(seed) # ------------------------------------------------------------------ # Public API # ------------------------------------------------------------------ def sample_evaporation( self, pion_ke_total_mev: float, n_events: int = 1, ) -> List[EvaporationEvent]: """ Sample nuclear evaporation following p̄-nucleus annihilation. Parameters ---------- pion_ke_total_mev : float Total kinetic energy of the pion shower (upper bound for E*). n_events : int Number of independent evaporation events to sample. Returns ------- List of EvaporationEvent """ events = [] for _ in range(n_events): E_star = self._sample_excitation(pion_ke_total_mev) events.append(self._evaporate(E_star)) return events # ------------------------------------------------------------------ # Private helpers # ------------------------------------------------------------------ def _sample_excitation(self, pion_ke_total_mev: float) -> float: """Sample nuclear excitation energy E* ~ Uniform(0, pion_ke_total).""" return self._rng.uniform(0.0, max(pion_ke_total_mev, 0.0)) def _temperature(self, E_star: float) -> float: """Nuclear temperature T = sqrt(E* / a), a = A/10 MeV⁻¹.""" A = self._nuc["A"] a = A / 10.0 return float(np.sqrt(max(E_star / a, 1e-10))) def _evaporate(self, E_star: float) -> EvaporationEvent: """Run the Weisskopf evaporation cascade.""" fragments_types = [] fragments_ke = [] fragments_dirs = [] nuc = self._nuc while E_star > _EVAP_THRESHOLD_MEV: T = self._temperature(E_star) # Compute relative emission widths for each channel widths = [] for ptype, mass_mev, sep_key, spin_d2s1 in _CHANNELS: sep = nuc[sep_key] if E_star <= sep: widths.append(0.0) continue available_ke = E_star - sep # Width ∝ (2s+1) * m * <sigma_inv> * T * exp(-sep/T) # Use simplified inverse cross-section: sigma_inv ~ pi*(r0*A^1/3)^2 width = float( spin_d2s1 * mass_mev * available_ke * np.exp(-sep / T) ) widths.append(max(width, 0.0)) total_width = sum(widths) if total_width <= 0.0: break # no energetically allowed channel # Choose emission channel by roulette u = self._rng.random() * total_width cumulative = 0.0 chosen_idx = 0 for idx, w in enumerate(widths): cumulative += w if u <= cumulative: chosen_idx = idx break ptype, mass_mev, sep_key, _ = _CHANNELS[chosen_idx] sep = nuc[sep_key] available_ke = E_star - sep # Sample fragment kinetic energy from Maxwell-Boltzmann: P(ε) ~ ε*exp(-ε/T) ke = self._sample_maxwell_boltzmann(available_ke, T) direction = self._isotropic_direction() fragments_types.append(ptype) fragments_ke.append(ke) fragments_dirs.append(direction) E_star -= (sep + ke) E_star = max(E_star, 0.0) # Residual excitation → γ deposit gamma_e = E_star if E_star > _GAMMA_THRESHOLD_MEV else 0.0 return EvaporationEvent( fragment_types=fragments_types, fragment_energies=fragments_ke, fragment_directions=fragments_dirs, gamma_energy_mev=gamma_e, ) def _sample_maxwell_boltzmann(self, max_ke: float, T: float) -> float: """ Sample KE from P(ε) ∝ ε·exp(-ε/T) via rejection sampling. Clipped to [0, max_ke]. """ for _ in range(200): # Proposal: exponential with mean T eps = self._rng.exponential(T) if eps > max_ke: continue # Accept with probability eps/T (normalised rejection envelope) if self._rng.random() < (eps / (T + 1e-30)): return float(eps) # Fallback: uniform sample return float(self._rng.uniform(0.0, max_ke)) def _isotropic_direction(self) -> np.ndarray: """Sample isotropic unit vector.""" theta = np.arccos(1.0 - 2.0 * self._rng.random()) phi = self._rng.uniform(0, 2 * np.pi) return np.array([ np.sin(theta) * np.cos(phi), np.sin(theta) * np.sin(phi), np.cos(theta), ]) def nuclear_fragment_dedx(ke_mev: np.ndarray, fragment_ptype: int, mat) -> np.ndarray: """ Heavy-ion Bethe-Bloch dE/dx [MeV/cm] for alpha particles and deuterons. Uses the Bethe formula with: - Effective charge z_eff = Z * (1 - exp(-125*beta / Z^(2/3))) (Barkas) - Fragment mass in place of electron mass Parameters ---------- ke_mev : np.ndarray Kinetic energy array [MeV]. fragment_ptype : int ALPHA (9) or DEUTERON (10). mat : Material Target material. Returns ------- np.ndarray dE/dx [MeV/cm] """ from src.particles import ALPHA_MASS_MEV, DEUTERON_MASS_MEV if fragment_ptype == ALPHA: Z_frag, mass = 2, ALPHA_MASS_MEV elif fragment_ptype == DEUTERON: Z_frag, mass = 1, DEUTERON_MASS_MEV else: raise ValueError(f"Unsupported fragment ptype {fragment_ptype}") T = np.maximum(ke_mev, 1e-10) E_total = T + mass p2 = E_total**2 - mass**2 beta2 = np.clip(p2 / E_total**2, 1e-10, 1.0 - 1e-10) beta = np.sqrt(beta2) # Barkas effective charge z_eff2 = (Z_frag * (1.0 - np.exp(-125.0 * beta / Z_frag**(2.0/3.0))))**2 gamma2 = 1.0 / (1.0 - beta2) K = 0.307075 # MeV cm^2 / mol I_mev = mat.mean_excitation_ev * 1e-6 Wmax = 2.0 * 0.511 * beta2 * gamma2 rho_mole = mat.density_g_cm3 / mat.A log_term = np.log(2.0 * 0.511 * beta2 * gamma2 * Wmax / I_mev**2) dedx = K * mat.Z * rho_mole * z_eff2 / beta2 * (0.5 * log_term - beta2) return np.maximum(dedx, 0.0)

""" rk4_vs_boris_accuracy.py ======================== Compares Boris and RK4 integrator accuracy in a non-uniform quadrupole field.
